<div className="mt-16 mb-16 max-w-5xl mx-auto px-6">
  <h2 className="text-2xl sm:text-3xl font-semibold text-gray-900 dark:text-white mb-3">
    Quick start
  </h2>
  <p className="text-base text-gray-600 dark:text-gray-400 mb-8">
    Get up and running with the Platform API in minutes
  </p>
  
  <Steps>
    <Step title="Clone and install dependencies">
      Clone the repository and install the required dependencies:
      
      ```bash
      git clone https://github.com/LeonardoCaero/platform-api.git
      cd platform-api
      npm install
      ```
    </Step>
    
    <Step title="Configure environment variables">
      Copy the example environment file and configure your database connection:
      
      ```bash
      cp .env.example .env
      ```
      
      <Note>
        Update the `.env` file with your PostgreSQL connection string and JWT secrets.
      </Note>
    </Step>
    
    <Step title="Initialize the database">
      Run database migrations and seed initial data:
      
      ```bash
      npm run db:rebuild
      ```
    </Step>
    
    <Step title="Start the development server">
      Launch the API server:
      
      ```bash
      npm run dev
      ```
      
      The API will be available at `http://localhost:4000`
    </Step>
  </Steps>
</div>
